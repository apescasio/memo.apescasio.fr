## Error Handling Best Practices

```javascript

async function robust_api_call(url, options = {}) {

            data: response.data,

    } catch (error) {

        if (error.code === 'ECONNABORTED') {

            error_type = 'dns';
        } else if (error.response) {

            error_message = 'Erreur r√©seau - aucune r√©ponse re√ßue';
            error_type = 'network';

            error: error_message,

            original_error: error
        };
    }
}
```
