## Context

Back in my high school days, I used to retype **again and again** my notes for Histoire Géo and other classes on the keyboard. 

Years later, I created this app to train my word per minute & my memory.

I type as fast as I can and while typing, I repeat each phrase in my brain. 

## Python App 

The base of the code comes from [Tech with Tim](https://www.youtube.com/watch?v=NQ5i1kJAA6Y), who I highly recommend to watch for Python projects to create.

Tim's code is available in his [GitHub](https://github.com/techwithtim/WPM_Typing_Test).

The changes I've done to the code were : 

- Remove the random selection of phrases in the txt file
- Possibility to escape after each phrase
- Keep track of each cycle
- Have an average of WPM after each cycle

```py title="train.py"
import curses
from curses import wrapper
import time
import random

def start_screen(stdscr):
	stdscr.clear()
	stdscr.addstr("Welcome to the Speed Typing Test!")
	stdscr.addstr("\nPress any key to begin!")
	stdscr.refresh()
	stdscr.getkey()

def display_text(stdscr, target, current, wpm=0):
	stdscr.addstr(target)
	stdscr.addstr(1, 0, f"WPM: {wpm}")

	for i, char in enumerate(current):
		correct_char = target[i]
		color = curses.color_pair(1)
		if char != correct_char:
			color = curses.color_pair(2)

		stdscr.addstr(0, i, char, color)

def load_text(line_index=0):
	with open("inputs.txt", "r") as f:
		lines = f.readlines()
		### Remove empty lines ###
		lines = [line.strip() for line in lines if line.strip()]
		if not lines:
			return "No text found in inputs.txt", 0
		### Return the line at the current index and the total number of lines ###
		return lines[line_index % len(lines)], len(lines)

def wpm_test(stdscr, line_index=0, cycle=1):
	target_text, total_lines = load_text(line_index)
	current_text = []
	wpm = 0
	start_time = time.time()
	stdscr.nodelay(True)

	while True:
		time_elapsed = max(time.time() - start_time, 1)
		wpm = round((len(current_text) / (time_elapsed / 60)) / 5)

		stdscr.clear()
		display_text(stdscr, target_text, current_text, wpm)
		current_line = (line_index % total_lines) + 1
		stdscr.addstr(2, 0, f"Cycle {cycle} - Line {current_line} of {total_lines}")
		stdscr.refresh()

		if "".join(current_text) == target_text:
			stdscr.nodelay(False)
			### Calculate final WPM for this phrase ###
			final_time = time.time() - start_time
			final_wpm = round((len(target_text) / (final_time / 60)) / 5)
			break

		try:
			key = stdscr.getkey()
		except:
			continue

		if ord(key) == 27:
			final_wpm = 0  # If escaped, no WPM recorded
			break

		if key in ("KEY_BACKSPACE", '\b', "\x7f"):
			if len(current_text) > 0:
				current_text.pop()
		elif len(current_text) < len(target_text):
			current_text.append(key)

	return line_index + 1, cycle, final_wpm


def main(stdscr):
	curses.init_pair(1, curses.COLOR_GREEN, curses.COLOR_BLACK)
	curses.init_pair(2, curses.COLOR_RED, curses.COLOR_BLACK)
	curses.init_pair(3, curses.COLOR_WHITE, curses.COLOR_BLACK)

	start_screen(stdscr)
	line_index = 0
	cycle = 1
	wpm_results = []  ### Store WPM for each phrase in current cycle ###
	cycle_averages = []  ### Store average WPM for each completed cycle ###

	### Get the total number of lines to know when to increment cycle ###
	_, total_lines = load_text(0)
	
	while True:
		line_index, cycle, phrase_wpm = wpm_test(stdscr, line_index, cycle)
		wpm_results.append(phrase_wpm)

		### Check if we've completed all lines and need to start a new cycle ###
		if line_index >= total_lines:
			### Calculate average for this cycle ###
			if wpm_results:
				cycle_average = round(sum(wpm_results) / len(wpm_results))
				cycle_averages.append(cycle_average)
			
			line_index = 0
			cycle += 1
			wpm_results = []  ### Reset for new cycle ###

		stdscr.clear()
		stdscr.addstr(0, 0, "You completed the text! Press any key to continue...")
		stdscr.addstr(1, 0, "Press ESC to quit.")

		### Show cycle averages if we have completed at least one cycle ###
		
		if cycle_averages:
			stdscr.addstr(3, 0, "Cycle Averages (WPM):")
			for i, avg in enumerate(cycle_averages):
				stdscr.addstr(4 + i, 0, f"  Cycle {i + 1}: {avg} WPM")
		
		stdscr.refresh()
		key = stdscr.getkey()
		
		if ord(key) == 27:
			break

wrapper(main)

```
